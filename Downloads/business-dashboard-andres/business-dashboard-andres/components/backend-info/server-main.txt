// CORS configuration - Must be before routes
app.use(cors({
  origin: function (origin, callback) {
    console.log('ðŸŒ CORS request from origin:', origin);
    
    // Allow requests with no origin (like mobile apps or curl requests)
    if (!origin) return callback(null, true);
    
    const allowedOrigins = [
      'http://localhost:3000',
      'http://127.0.0.1:3000',
      'http://localhost:3001',
      'http://127.0.0.1:3001',
      process.env.SOCKET_CORS_ORIGIN || 'http://localhost:3000'
    ];
    
    if (allowedOrigins.indexOf(origin) !== -1) {
      console.log('âœ… CORS origin allowed:', origin);
      callback(null, true);
    } else {
      console.log('âŒ CORS origin blocked:', origin);
      callback(new Error('Not allowed by CORS'));
    }
  },
  credentials: true,
  methods: ['GET', 'POST', 'PUT', 'DELETE', 'OPTIONS', 'PATCH'],
  allowedHeaders: ['Content-Type', 'Authorization', 'X-Requested-With', 'Accept', 'Origin'],
  optionsSuccessStatus: 200,
  preflightContinue: false,
  // Ensure CORS headers are always sent
  maxAge: 86400 // Cache preflight for 24 hours
}));

// Handle preflight requests explicitly
app.options('*', (req, res) => {
  console.log('ðŸ”„ Preflight request received');
  res.header('Access-Control-Allow-Origin', req.headers.origin || 'http://localhost:3000');
  res.header('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE, OPTIONS, PATCH');
  res.header('Access-Control-Allow-Headers', 'Content-Type, Authorization, X-Requested-With, Accept, Origin');
  res.header('Access-Control-Allow-Credentials', 'true');
  res.header('Access-Control-Max-Age', '86400');
  res.status(200).end();
});

    res.header('Access-Control-Allow-Credentials', 'true');
  }
  next();
});

// Socket.IO connection handling
const connectedUsers = new Map();
const connectedDrivers = new Map();

io.on('connection', (socket) => {
  console.log('New client connected:', socket.id);

  // Authenticate user
  socket.on('authenticate', async (data) => {
    try {
      const jwt = require('jsonwebtoken');
      const Company = require('./models/Company');
      
      const decoded = jwt.verify(data.token, process.env.JWT_SECRET);
      const company = await Company.findById(decoded.companyId);
      
      if (company) {
        socket.companyId = company._id;
        socket.userType = 'company';
        connectedUsers.set(socket.id, {
          companyId: company._id,
          socketId: socket.id,
          userType: 'company'
        });
        
        socket.emit('authenticated', { 
          message: 'Authentication successful',
          companyId: company._id 
        });
        
        console.log(`Company ${company.name} authenticated on socket ${socket.id}`);
      } else {
        socket.emit('auth_error', { message: 'Invalid token' });
      }
    } catch (error) {
      socket.emit('auth_error', { message: 'Authentication failed' });
    }
  });

  // Driver authentication
  socket.on('driver_authenticate', async (data) => {
    try {
      const driver = await Driver.findOne({ 
        phone: data.phone, 
        password: data.password,
        isActive: true 
      });
      
      if (driver) {
        socket.driverId = driver._id;
        socket.userType = 'driver';
        connectedDrivers.set(socket.id, {
          driverId: driver._id,
          socketId: socket.id,
          userType: 'driver',
          currentLocation: driver.currentLocation
        });
        
        socket.emit('driver_authenticated', { 
          message: 'Driver authentication successful',
          driverId: driver._id 
        });
        
        console.log(`Driver ${driver.name} authenticated on socket ${socket.id}`);
      } else {
        socket.emit('auth_error', { message: 'Invalid driver credentials' });
      }
    } catch (error) {
      socket.emit('auth_error', { message: 'Driver authentication failed' });
    }
  });

  // Update driver location
  socket.on('update_location', async (data) => {
    try {
      if (socket.userType !== 'driver' || !socket.driverId) {
        return socket.emit('error', { message: 'Unauthorized' });
      }

      const { lat, lng } = data;
      
      // Update driver location in database
      await Driver.findByIdAndUpdate(socket.driverId, {
        currentLocation: {
          lat,
          lng,
          lastUpdated: new Date()
        }
      });

      // Update in connected drivers map
      const driverInfo = connectedDrivers.get(socket.id);
      if (driverInfo) {
        driverInfo.currentLocation = { lat, lng, lastUpdated: new Date() };
        connectedDrivers.set(socket.id, driverInfo);
      }

      // Notify company about driver location update
      const companySocket = Array.from(connectedUsers.values())
        .find(user => user.companyId === driverInfo?.companyId);
      
      if (companySocket) {
        io.to(companySocket.socketId).emit('driver_location_updated', {
          driverId: socket.driverId,
          location: { lat, lng }
        });
      }

    } catch (error) {
      console.error('Update location error:', error);
      socket.emit('error', { message: 'Failed to update location' });
    }
  });

  // New order notification
  socket.on('new_order', async (data) => {
    try {
      if (socket.userType !== 'company' || !socket.companyId) {
        return socket.emit('error', { message: 'Unauthorized' });
      }

      // Emit to all connected drivers for this company
      const companyDrivers = Array.from(connectedDrivers.values())
        .filter(driver => driver.companyId === socket.companyId);
      
      companyDrivers.forEach(driver => {
        io.to(driver.socketId).emit('new_order_available', data);
      });

    } catch (error) {
      console.error('New order notification error:', error);
      socket.emit('error', { message: 'Failed to notify drivers' });
    }
  });

  // Order status update
  socket.on('order_status_update', async (data) => {
    try {
      const { orderId, status, driverId } = data;
      
      // Update order in database
      const order = await Order.findByIdAndUpdate(orderId, {
        status,
        driverId: driverId || null,
        ...(status === 'delivered' && { actualDeliveryTime: new Date() })
      }, { new: true });

      if (!order) {
        return socket.emit('error', { message: 'Order not found' });
      }

      // Notify company about status update
      const companySocket = Array.from(connectedUsers.values())
        .find(user => user.companyId === order.companyId);
      
      if (companySocket) {
        io.to(companySocket.socketId).emit('order_status_changed', {
          orderId,
          status,
          order
        });
      }

      // Notify customer if they're connected
      socket.broadcast.emit('order_status_changed', {
        orderId,
        status,
        order
      });

    } catch (error) {
      console.error('Order status update error:', error);
      socket.emit('error', { message: 'Failed to update order status' });
    }
  });

  // Moto request handling
  socket.on('new_moto_request', async (data) => {
    try {
      if (socket.userType !== 'company' || !socket.companyId) {
        return socket.emit('error', { message: 'Unauthorized' });
      }

      // Emit to all available drivers
      const availableDrivers = Array.from(connectedDrivers.values())
        .filter(driver => driver.status === 'available');
      
      availableDrivers.forEach(driver => {
        io.to(driver.socketId).emit('new_moto_request', data);
      });

    } catch (error) {
      console.error('New moto request error:', error);
      socket.emit('error', { message: 'Failed to process moto request' });
    }
  });

  // Moto request status update
  socket.on('moto_status_update', async (data) => {
    try {
      const { requestId, status, driverId } = data;
      
      const motoRequest = await MotoRequest.findByIdAndUpdate(requestId, {
        status,
        driverId: driverId || null,
        ...(status === 'pickup' && { actualPickupTime: new Date() }),
        ...(status === 'completed' && { actualDeliveryTime: new Date() })
      }, { new: true });

      if (!motoRequest) {
        return socket.emit('error', { message: 'Moto request not found' });
      }

      // Notify company
      const companySocket = Array.from(connectedUsers.values())
        .find(user => user.companyId === motoRequest.companyId);
      
      if (companySocket) {
        io.to(companySocket.socketId).emit('moto_status_changed', {
          requestId,
          status,
          motoRequest
        });
      }

    } catch (error) {
      console.error('Moto status update error:', error);
      socket.emit('error', { message: 'Failed to update moto status' });
    }
  });

  // Product variable updates
  socket.on('productVariableUpdate', async (data) => {
    try {
      if (socket.userType !== 'company' || !socket.companyId) {
        return socket.emit('error', { message: 'Unauthorized' });
      }

      const { productId, variableId, action } = data;
      
      // Broadcast to all connected clients of this company
      const companyUsers = Array.from(connectedUsers.values())
        .filter(user => user.companyId === socket.companyId);
      
      companyUsers.forEach(user => {
        io.to(user.socketId).emit('productVariableChanged', {
          productId,
          variableId,
          action,
          timestamp: new Date().toISOString()
        });
      });
      
      console.log(`Product variable ${action}:`, { productId, variableId });
    } catch (error) {
      console.error('Error handling product variable update:', error);
      socket.emit('error', { message: 'Failed to process variable update' });
    }
  });

  // Pricing option updates
  socket.on('pricingOptionUpdate', async (data) => {
    try {
      if (socket.userType !== 'company' || !socket.companyId) {
        return socket.emit('error', { message: 'Unauthorized' });
      }

      const { variableId, optionId, action } = data;
      
      // Broadcast to all connected clients of this company
      const companyUsers = Array.from(connectedUsers.values())
        .filter(user => user.companyId === socket.companyId);
      
      companyUsers.forEach(user => {
        io.to(user.socketId).emit('pricingOptionChanged', {
          variableId,
          optionId,
          action,
          timestamp: new Date().toISOString()
        });
      });
      
      console.log(`Pricing option ${action}:`, { variableId, optionId });
    } catch (error) {
      console.error('Error handling pricing option update:', error);
      socket.emit('error', { message: 'Failed to process pricing option update' });
    }
  });

  // Disconnect handling
  socket.on('disconnect', () => {
    console.log('Client disconnected:', socket.id);
    
    // Remove from connected users/drivers
    connectedUsers.delete(socket.id);
    connectedDrivers.delete(socket.id);
  });
});

// Initialize database and start server
const startServer = async () => {
  try {
    // Connect to MongoDB
    await connectDB();
    
    // Initialize MinIO
    await initializeMinio();
    
    const PORT = process.env.PORT || 3001;
    
    server.listen(PORT, () => {
      console.log(`Server running on port ${PORT}`);
      console.log(`Environment: ${process.env.NODE_ENV}`);
      console.log(`Socket.IO enabled`);
      console.log(`MinIO initialized`);
    });
    
  } catch (error) {
    console.error('Failed to start server:', error);
    process.exit(1);
  }
};
